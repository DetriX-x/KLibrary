 SELECT * FROM (SELECT 
 	b.book_id,
    b.title,
    array_to_string(array_agg((au.name::text || ' '::text) || au.surname::text), ', '::text) AS array_to_string,
    ge.genre_name,
    b.lang,
    b.year,
    e.edition_name,
    b.quantity AS quant,
    b.isbn
 FROM books b
     LEFT JOIN books_authors ba ON ba.book_id = b.book_id
     LEFT JOIN authors au USING (author_id)
     JOIN genres ge ON b.genre_id = ge.genre_id
     JOIN editions e ON b.edition_id = e.edition_id
 GROUP BY b.book_id, b.title, ge.genre_name, b.lang, b.year, e.edition_name, b.quantity, b.isbn) AS foo
 WHERE foo.quant < 10
